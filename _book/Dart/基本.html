void start() {
  print('Hello world!');
  String name = 'coderwhy';
  int age = 18;
  double height = 1.88;
  print('${name}, ${age}, ${height}');
}

void runtimeType() {
  // var/dynamic/const/final 变量名称 = 赋值;
  var name = 'coderwhy';
  name = 'kobe';
  print(name.runtimeType); // String
}

dynamicType() {
  // 不推荐 
  dynamic name = 'coderwhy';
  print(name.runtimeType); // String
  name = 18;
  print(name.runtimeType); // int
}

finalAndConst() {
  final name = 'coderwhy';
  // name = 'kobe'; // 错误做法
  const age = 18;
  // age = 20; // 错误做法
}

finalAndConstDifference() {
  // const在赋值时, 赋值的内容必须是在编译期间就确定下来的
  // final在赋值时, 可以动态获取, 比如赋值一个函数
  // const name = getName(); // 错误的做法, 因为要执行函数才能获取到值
  final name = getName(); // 正确的做法
  print(name);
}

String getName() {
  return 'coderwhy';
}

void boolDemo() {
  var isFlag = true;
  print('$isFlag ${isFlag.runtimeType}'); // true bool
  // 注意: Dart中不能判断非0即真, 或者非空即真
  var message = 'Hello Dart';
  // 错误的写法
  /**
  if (message) {
    print(message)
  }
   */
}

void strDemo() {
  // Dart字符串是UTF-16编码单元的序列。您可以使用单引号或双引号创建一个字符串
  // 可以使用三个单引号或者双引号表示多行字符串:
  var message1 = '''
  哈哈哈
  呵呵呵
  嘿嘿嘿''';
  print(message1);
  var message2 = "一山更比一山高";
  print(message2);
}

void strConcatDemo() {
  // 字符串和其他变量或表达式拼接: 使用${expression}, 如果表达式是一个标识符, 那么{}可以省略
  var name = 'coderwhy';
  var age = 18;
  var height = 1.88;
  print('my name is ${name}, age is $age, height is $height');
}
